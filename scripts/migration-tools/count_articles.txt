import { getPayload } from 'payload';
import configPromise from './src/payload.config';

async function run() {
  const payload = await getPayload({ config: configPromise });
  const published = await payload.find({ collection: 'articles', limit: 1 });
  const drafts = await payload.find({ collection: 'articles', draft: true, limit: 1 });
  
  console.log(`Published totalDocs: ${published.totalDocs}`);
  console.log(`Including Drafts totalDocs: ${drafts.totalDocs}`);
  process.exit(0);
}
run();
